EXPERIMENT-4

AIM

Write a program in to implement Huffman Coding and analyse its time complexity


ALGORITHM

1.	Input: A set of characters and their frequencies.
2.	Build a Min-Heap (priority queue) of all characters with their frequencies.
3.	While the heap has more than one node:
•	Extract the two nodes with the lowest frequency.
•	Create a new internal node with frequency equal to the sum of the two nodes.
•	Set the extracted nodes as the left and right children of the new node.
•	Insert the new node back into the heap.
4.	The remaining node in the heap is the root of the Huffman Tree.
5.	Traverse the tree:
•	Assign 0 to the left edge and 1 to the right edge.
•	Store the generated binary code for each character.



CODE

#include <iostream>
#include <queue>
#include <unordered_map>
using namespace std;

// Tree node
struct Node {
    char ch;
    int freq;
    Node *left, *right;

    Node(char c, int f) {
        ch = c;
        freq = f;
        left = right = nullptr;
    }
};

// Custom comparator for min-heap
struct compare {
    bool operator()(Node* l, Node* r) {
        return l->freq > r->freq;
    }
};

// Function to print Huffman Codes
void printCodes(Node* root, string str) {
    if (!root) return;

    if (root->ch != '$') // '$' means internal node
        cout << root->ch << ": " << str << "\n";

    printCodes(root->left, str + "0");
    printCodes(root->right, str + "1");
}

// Build Huffman Tree
void HuffmanCoding(char data[], int freq[], int n) {
    priority_queue<Node*, vector<Node*>, compare> minHeap;

    // Step 1: Create leaf nodes
    for (int i = 0; i < n; i++)
        minHeap.push(new Node(data[i], freq[i]));

    // Step 2: Build tree
    while (minHeap.size() > 1) {
        Node* left = minHeap.top(); minHeap.pop();
        Node* right = minHeap.top(); minHeap.pop();

        Node* newNode = new Node('$', left->freq + right->freq);
        newNode->left = left;
        newNode->right = right;

        minHeap.push(newNode);
    }

    // Step 3: Print Huffman Codes
    printCodes(minHeap.top(), "");
}

// Driver Code
int main() {
    char arr[] = { 'a', 'b', 'c', 'd', 'e', 'f' };
    int freq[] = { 5, 9, 12, 13, 16, 45 };
    int n = sizeof(arr) / sizeof(arr[0]);

    HuffmanCoding(arr, freq, n);
    return 0;
}


OUTPUT:
 


TIME COMPLEXITY:

1.	Heap construction: O (n)
2.	Heap operations (extract + insert): O (n log n)
3.	Tree traversal (for code generation): O (n)

Overall Time Complexity: O (n log n)
