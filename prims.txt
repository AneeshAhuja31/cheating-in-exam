#include <iostream>
#include <vector>
#include <climits>
using namespace std;

struct Edge
{
    int src, dest, weight;
};

void primsMST(vector<Edge> &edges, int V)
{
    vector<vector<pair<int, int>>> adj(V);

    // Build adjacency list
    for (auto &e : edges)
    {
        adj[e.src].push_back({e.dest, e.weight});
        adj[e.dest].push_back({e.src, e.weight});
    }

    vector<int> key(V, INT_MAX);
    vector<int> parent(V, -1);
    vector<bool> inMST(V, false);

    key[0] = 0;

    for (int count = 0; count < V - 1; count++)
    {
        int u = -1, minKey = INT_MAX;

        // Pick vertex with minimum key not in MST
        for (int v = 0; v < V; v++)
        {
            if (!inMST[v] && key[v] < minKey)
            {
                minKey = key[v];
                u = v;
            }
        }

        inMST[u] = true;

        // Update neighbors
        for (auto &edge : adj[u])
        {
            int v = edge.first;
            int w = edge.second;
            if (!inMST[v] && w < key[v])
            {
                key[v] = w;
                parent[v] = u;
            }
        }
    }

    int mstWeight = 0;
    cout << "Edges in the Minimum Spanning Tree:\n";
    for (int i = 1; i < V; i++)
    {
        cout << parent[i] << " -- " << i << " => " << key[i] << endl;
        mstWeight += key[i];
    }

    cout << "Total weight of MST: " << mstWeight << endl;
}

int main()
{
    int V = 4;
    vector<Edge> edges = {
        {0, 1, 10},
        {0, 2, 6},
        {0, 3, 5},
        {1, 3, 15},
        {2, 3, 4}};

    primsMST(edges, V);

    return 0;
}
