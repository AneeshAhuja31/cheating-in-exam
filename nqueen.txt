Program:- 8

AIM:-
Write a program to implement N Queen's problem using Back Tracking.

ALGORITHM:-
1.	Start from the firstrow and try to place a queen in the first column.
2.	Check if placing the queen at the current position is safe:
o	No other queen is in the same column.
o	No other queen is in the same left diagonal.
o	No other queen is in the same right diagonal.
3.	If it is safe, place the queen and move to the next row recursively.
4.	If placing a queen in any column of the current row is not possible, backtrack to the previous row and move the queen to the next column.
5.	Repeat until all queens are placed or all configurations are exhausted.

CODE:-
#include <iostream>
#include <vector>

using namespace std;

// Checks if placing a queen at board[row][col] is safe
bool isSafe(const vector<string>& board, int row, int col, int n) {
    
    // Check column above
    for (int i = 0; i < row; i++) {
        if (board[i][col] == 'Q')
            return false;
    }

    // Check upper-left diagonal
    for (int i = row - 1, j = col - 1; i >= 0 && j >= 0; i--, j--) {
        if (board[i][j] == 'Q')
            return false;
    }

    // Check upper-right diagonal
    for (int i = row - 1, j = col + 1; i >= 0 && j < n; i--, j++) {
        if (board[i][j] == 'Q')
            return false;
    }

    return true;
}

// Backtracking function
bool solveNQueensUtil(vector<string>& board, int row, int n) {
    
    // All queens placed
    if (row == n)
        return true;

    // Try placing a queen at each column
    for (int col = 0; col < n; col++) {
        
        if (isSafe(board, row, col, n)) {
            
            board[row][col] = 'Q';   // Place queen

            if (solveNQueensUtil(board, row + 1, n))
                return true;

            board[row][col] = '.';   // Backtrack
        }
    }

    return false;  // No valid placement found
}

// Main solver function
void solveNQueens(int n) {
    vector<string> board(n, string(n, '.'));

    if (solveNQueensUtil(board, 0, n)) {
        for (const auto& row : board) {
            cout << row << "\n";
        }
    } else {
        cout << "No solution exists.\n";
    }
}

int main() {
    int n;
    cout << "Enter the number of queens: ";
    cin >> n;

    solveNQueens(n);

    return 0;
}

TIME COMPLEXITY:-
•	The worst-case time complexity of the N-Queens backtracking solution is O(N!).
•	For the first row, there are N choices.
•	For the second row, there are at most (N-1) choices (since you cannot place a queen in the same column as the first queen).
•	For the third row, there are at most (N-2) choices, and so forth.
Hence, the total number of possible configurations is roughly N * (N-1) * (N-2) * ... * 1 = N!.
