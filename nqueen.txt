#include <iostream>
#include <vector>

using namespace std;

boolisSafe(const vector<string>& board, int row, int col, int n) {
for (inti = 0; i< row; i++) {
if (board[i][col] == 'Q')
return false;
    }

for (inti = row - 1, j = col - 1; i>= 0 && j >= 0; i--, j--) {
if (board[i][j] == 'Q')
return false;
    }

for (inti = row - 1, j = col + 1; i>= 0 && j < n; i--, j++) {
if (board[i][j] == 'Q')
return false;
    }

return true;
}

boolsolveNQueensUtil(vector<string>& board, int row, int n) {
if (row == n) {
        // All queens placed successfully
return true;
    }

for (int col = 0; col < n; col++) {
if (isSafe(board, row, col, n)) {
board[row][col] = 'Q';

if (solveNQueensUtil(board, row + 1, n))
return true;
board[row][col] = '.';
        }
    }

return false;
}

voidsolveNQueens(int n) {
vector<string> board(n, string(n, '.'));

if (solveNQueensUtil(board, 0, n)) {
for (const auto& row : board) {
cout<< row << "\n";
        }
    } else {
cout<< "No solution exists.\n";
    }
}

int main() {
int n;
cout<< "Enter the number of queens: ";
cin>> n;
solveNQueens(n);
return 0;
}
