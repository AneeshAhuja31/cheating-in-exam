#include <iostream>
#include <string>
#include <vector>
using namespace std;

const int d = 256; // number of characters in input alphabet
const int q = 101; // a prime number for Rabin-Karp

// 1. Naive search
void naiveSearch(const string &txt, const string &pat) {
cout<< "--- Naive Search ---\n";
    int M = (int)pat.size();
    int N = (int)txt.size();
    for (int i = 0; i<= N - M; ++i) {
        int j;
        for (j = 0; j < M; ++j)
            if (txt[i + j] != pat[j])
break;
        if (j == M)
cout<< "Pattern found at index " <<i<< '\n';
    }
}

// 2. Rabin-Karp
void rabinKarpSearch(const string &txt, const string &pat) {
cout<< "--- Rabin-Karp Search ---\n";
    int M = (int)pat.size();
    int N = (int)txt.size();
    if (M > N) return;

    long long p = 0; // hash value for pattern
    long long t = 0; // hash value for txt window
    long long h = 1;

    for (int i = 0; i< M - 1; ++i)
        h = (h * d) % q;

    for (int i = 0; i< M; ++i) {
        p = (d * p + static_cast<unsigned char>(pat[i])) % q;
        t = (d * t + static_cast<unsigned char>(txt[i])) % q;
    }

    for (int i = 0; i<= N - M; ++i) {
        if (p == t) {
            int j;
            for (j = 0; j < M; ++j)
                if (txt[i + j] != pat[j])
break;
            if (j == M)
cout<< "Pattern found at index " <<i<< '\n';
        }
        if (i< N - M) {
            t = (d * (t - static_cast<unsigned char>(txt[i]) * h) + static_cast<unsigned char>(txt[i + M])) % q;
            if (t < 0) t += q;
        }
    }
}

// helper for KMP
void computeLPSArray(const string &pat, vector<int>&lps) {
    int M = (int)pat.size();
    int len = 0;
lps[0] = 0;
    int i = 1;
    while (i< M) {
        if (pat[i] == pat[len]) {
            ++len;
lps[i] = len;
            ++i;
        } else {
            if (len != 0) {
len = lps[len - 1];
            } else {
lps[i] = 0;
                ++i;
            }
        }
    }
}

// 3. KMP
void KMPSearch(const string &txt, const string &pat) {
cout<< "--- KMP Search ---\n";
    int M = (int)pat.size();
    int N = (int)txt.size();
    if (M == 0) return;

    vector<int>lps(M);
computeLPSArray(pat, lps);

    int i = 0; // index for txt
    int j = 0; // index for pat
    while (i< N) {
        if (pat[j] == txt[i]) {
            ++i;
            ++j;
        }
        if (j == M) {
cout<< "Pattern found at index " << (i - j) << '\n';
            j = lps[j - 1];
        } else if (i< N && pat[j] != txt[i]) {
            if (j != 0)
                j = lps[j - 1];
            else
                ++i;
        }
    }
}

int main() {
    string txt = "ABABDABACDABABCABAB";
    string pat = "ABABCABAB";

cout<< "Text: " << txt << '\n';
cout<< "Pattern: " << pat << "\n\n";

naiveSearch(txt, pat);
cout<< '\n';
rabinKarpSearch(txt, pat);
cout<< '\n';
KMPSearch(txt, pat);

    return 0;
}
