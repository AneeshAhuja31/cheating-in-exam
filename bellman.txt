#include <iostream>
#include <vector>
using namespace std;

vector<int> bellmanFord(int V, vector<vector<int>>& edges, int src) {

    // Initially distance from source to all other vertices is infinite.
    vector<int> dist(V, 1e8);
    dist[src] = 0;

    // Relax all edges V times (extra iteration to detect negative cycle)
    for (int i = 0; i < V; i++) {

        for (vector<int> edge : edges) {
            int u = edge[0];
            int v = edge[1];
            int wt = edge[2];
            
            if (dist[u] != 1e8 && dist[u] + wt < dist[v]) {

                // If this is the Vth relaxation, there is a negative cycle
                if (i == V - 1)
                    return {-1};

                // Update shortest distance to node v
                dist[v] = dist[u] + wt;
            }
        }
    }

    return dist;
}

int main() {

    // Number of vertices
    int V = 5;

    // Edge list: {source, destination, weight} using 0-based numbering
    vector<vector<int>> edges = {
        {0, 1, 5}, 
        {0, 2, 3}, 
        {1, 2, 2}, 
        {1, 3, 6}, 
        {2, 3, 7},
        {2, 4, 4},
        {3, 4, -1}
    };

    // Source vertex
    int src = 0;

    // Run Bellman-Ford
    vector<int> ans = bellmanFord(V, edges, src);

    // Output distances
    cout << "Path\t: Shortest Distance" << endl;
    cout << "---------------------------" << endl;
    for (int x = 0; x < ans.size(); x++)
        cout << src << " --> " << x << " : " << ans[x] << endl;

    return 0;
}
